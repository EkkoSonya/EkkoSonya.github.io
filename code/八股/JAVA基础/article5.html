                                              жіЁи§Ј Annotation

                                              жіЁи§ЈпјҲAnnotationпјүжҳҜ Java жҸҗдҫӣзҡ„дёҖз§Қе…ғж•°жҚ®жңәеҲ¶пјҢз”ЁдәҺдёәд»Јз Ғж·»еҠ иҜҙжҳҺдҝЎжҒҜпјҢеҸҜд»ҘеңЁзј–иҜ‘жңҹгҖҒзұ»еҠ иҪҪжңҹжҲ–иҝҗиЎҢжңҹиў«иҜ»еҸ–е’ҢеӨ„зҗҶ

                                              Java иҜӯиЁҖдёӯзҡ„зұ»гҖҒж–№жі•гҖҒеҸҳйҮҸгҖҒеҸӮж•°е’ҢеҢ…зӯүйғҪеҸҜд»Ҙиў«ж ҮжіЁпјҢJava ж ҮжіЁеҸҜд»ҘйҖҡиҝҮеҸҚе°„иҺ·еҸ–ж ҮжіЁеҶ…е®№гҖӮеңЁзј–иҜ‘еҷЁз”ҹжҲҗзұ»ж–Үд»¶ж—¶пјҢж ҮжіЁеҸҜд»Ҙиў«еөҢе…ҘеҲ°еӯ—иҠӮз ҒдёӯпјҢJava иҷҡжӢҹжңәеҸҜд»Ҙдҝқз•ҷж ҮжіЁеҶ…е®№пјҢеңЁиҝҗиЎҢж—¶еҸҜд»ҘиҺ·еҸ–еҲ°ж ҮжіЁеҶ…е®№

                                              жіЁи§Јжң¬иҙЁдёҠжҳҜдёҖз§Қзү№ж®Ҡзҡ„жҺҘеҸЈпјҢе®ғз»§жүҝиҮӘ java.lang.annotation.Annotation жҺҘеҸЈпјҢжүҖд»ҘжіЁи§Јд№ҹеҸ«еЈ°жҳҺејҸжҺҘеҸЈпјҢзј–иҜ‘еҗҺпјҢJava зј–иҜ‘еҷЁдјҡе°Ҷе…¶иҪ¬жҚўдёәдёҖдёӘз»§жүҝиҮӘ Annotation зҡ„жҺҘеҸЈпјҢе№¶з”ҹжҲҗзӣёеә”зҡ„еӯ—иҠӮз Ғж–Үд»¶

                                              // е®ҡд№үжіЁи§Ј
public @interface MyAnnotation {
  String value();
}

// зј–иҜ‘еҗҺе®һйҷ…дёҠжҳҜпјҡ
public interface MyAnnotation extends java.lang.annotation.Annotation {
  String value();
}

package com.ekko;

import java.lang.annotation.ElementType;
import java.lang.annotation.Retention;
import java.lang.annotation.RetentionPolicy;
import java.lang.annotation.Target;

@Target({ElementType.TYPE, ElementType.METHOD})
@Retention(RetentionPolicy.RUNTIME)
public @interface Penguin {
    String value() default "";
}

// com/ekko/Main.java
package com.ekko;

@Penguin("penguin")
public class Main {
    @Penguin("test")
    public static void main(String[] args) throws InterruptedException {
        System.out.println("penguin");
    }
}

                                              Annotation зҡ„з»„жҲҗдёӯпјҢжңү 3 дёӘйқһеёёйҮҚиҰҒзҡ„дё»е№Ізұ»пјҢAnnotation жҺҘеҸЈзҲ¶зұ»пјҢElementType еҶіе®ҡдҪңз”ЁеҹҹпјҢRetentionPolicy з”ҹж•ҲиҢғеӣҙ

                                              Annotation е°ұжҳҜдёӘжҺҘеҸЈ, жҜҸ 1 дёӘ Annotation еҜ№иұЎпјҢйғҪдјҡжңүе”ҜдёҖзҡ„ RetentionPolicy еұһжҖ§пјӣиҮідәҺ ElementType еұһжҖ§пјҢеҲҷжңү 1~n дёӘ

                                              • ElementType жҳҜ Enum жһҡдёҫзұ»еһӢпјҢе®ғз”ЁжқҘжҢҮе®ҡ Annotation зҡ„зұ»еһӢ
                                              • RetentionPolicy жҳҜ Enum жһҡдёҫзұ»еһӢпјҢе®ғз”ЁжқҘжҢҮе®ҡ Annotation зҡ„зӯ–з•Ҙ
                                                • иӢҘ Annotation зҡ„зұ»еһӢдёә SOURCEпјҢеҲҷж„Ҹе‘ізқҖпјҡAnnotation д»…еӯҳеңЁдәҺзј–иҜ‘еҷЁеӨ„зҗҶжңҹй—ҙпјҢзј–иҜ‘еҷЁеӨ„зҗҶе®Ңд№ӢеҗҺпјҢиҜҘ Annotation е°ұжІЎз”ЁдәҶ
                                                • иӢҘ Annotation зҡ„зұ»еһӢдёә CLASSпјҢеҲҷж„Ҹе‘ізқҖпјҡзј–иҜ‘еҷЁе°Ҷ Annotation еӯҳеӮЁдәҺзұ»еҜ№еә”зҡ„ .class ж–Үд»¶дёӯпјҢе®ғжҳҜ Annotation зҡ„й»ҳи®ӨиЎҢдёә
                                                • иӢҘ Annotation зҡ„зұ»еһӢдёә RUNTIMEпјҢеҲҷж„Ҹе‘ізқҖпјҡзј–иҜ‘еҷЁе°Ҷ Annotation еӯҳеӮЁдәҺ class ж–Үд»¶дёӯпјҢе№¶дё”еҸҜз”ұJVMиҜ»е…Ҙ

                                              RetentionPolicy дҪңз”ЁиҢғеӣҙ
                                              package java.lang.annotation;
public enum RetentionPolicy {
    SOURCE,            /* AnnotationдҝЎжҒҜд»…еӯҳеңЁдәҺзј–иҜ‘еҷЁеӨ„зҗҶжңҹй—ҙпјҢзј–иҜ‘еҷЁеӨ„зҗҶе®Ңд№ӢеҗҺе°ұжІЎжңүиҜҘAnnotationдҝЎжҒҜдәҶ  */
    CLASS,             /* зј–иҜ‘еҷЁе°ҶAnnotationеӯҳеӮЁдәҺзұ»еҜ№еә”зҡ„.classж–Үд»¶дёӯгҖӮй»ҳи®ӨиЎҢдёә  */
    RUNTIME            /* зј–иҜ‘еҷЁе°ҶAnnotationеӯҳеӮЁдәҺclassж–Үд»¶дёӯпјҢе№¶дё”еҸҜз”ұJVMиҜ»е…Ҙ */
}
                                              • жәҗз Ғзә§еҲ«жіЁи§Ј пјҡд»…еӯҳеңЁдәҺжәҗз ҒдёӯпјҢзј–иҜ‘еҗҺдёҚдјҡдҝқз•ҷпјҲ@Retention(RetentionPolicy.SOURCE)пјүгҖӮ
                                              • зұ»ж–Үд»¶зә§еҲ«жіЁи§Ј пјҡдҝқз•ҷеңЁ .class ж–Үд»¶дёӯпјҢдҪҶиҝҗиЎҢж—¶дёҚеҸҜи§ҒпјҲ@Retention(RetentionPolicy.CLASS)пјүгҖӮ
                                              • иҝҗиЎҢж—¶жіЁи§Ј пјҡдҝқз•ҷеңЁ .class ж–Үд»¶дёӯпјҢе№¶дё”еҸҜд»ҘйҖҡиҝҮеҸҚе°„еңЁиҝҗиЎҢж—¶и®ҝй—®пјҲ@Retention(RetentionPolicy.RUNTIME)пјү

                                              еҸӘжңүиҝҗиЎҢж—¶жіЁи§ЈеҸҜд»ҘйҖҡиҝҮеҸҚе°„жңәеҲ¶иҝӣиЎҢи§Јжһҗ

                                              ElementType дҪңз”Ёеҹҹ
                                              package java.lang.annotation;

public enum ElementType {
    TYPE,               /* зұ»гҖҒжҺҘеҸЈпјҲеҢ…жӢ¬жіЁйҮҠзұ»еһӢпјүжҲ–жһҡдёҫеЈ°жҳҺ  */
    FIELD,              /* еӯ—ж®өеЈ°жҳҺпјҲеҢ…жӢ¬жһҡдёҫеёёйҮҸпјү  */
    METHOD,             /* ж–№жі•еЈ°жҳҺ  */
    PARAMETER,          /* еҸӮж•°еЈ°жҳҺ  */
    CONSTRUCTOR,        /* жһ„йҖ ж–№жі•еЈ°жҳҺ  */
    LOCAL_VARIABLE,     /* еұҖйғЁеҸҳйҮҸеЈ°жҳҺ  */
    ANNOTATION_TYPE,    /* жіЁйҮҠзұ»еһӢеЈ°жҳҺ  */
    PACKAGE             /* еҢ…еЈ°жҳҺ  */
}

                                              жіЁи§Јзҡ„дҪңз”ЁеҹҹпјҲScopeпјүжҢҮзҡ„жҳҜжіЁи§ЈеҸҜд»Ҙеә”з”ЁеңЁе“ӘдәӣзЁӢеәҸе…ғзҙ дёҠпјҢдҫӢеҰӮзұ»гҖҒж–№жі•гҖҒеӯ—ж®өзӯү

                                              • зұ»зә§еҲ«дҪңз”Ёеҹҹпјҡз”ЁдәҺжҸҸиҝ°зұ»зҡ„жіЁи§ЈпјҢйҖҡеёёж”ҫзҪ®еңЁзұ»е®ҡд№үзҡ„дёҠйқўпјҢеҸҜд»Ҙз”ЁжқҘжҢҮе®ҡзұ»зҡ„дёҖдәӣеұһжҖ§пјҢеҰӮзұ»зҡ„и®ҝй—®зә§еҲ«гҖҒз»§жүҝе…ізі»гҖҒжіЁйҮҠзӯүгҖӮ
                                              • ж–№жі•зә§еҲ«дҪңз”Ёеҹҹпјҡз”ЁдәҺжҸҸиҝ°ж–№жі•зҡ„жіЁи§ЈпјҢйҖҡеёёж”ҫзҪ®еңЁж–№жі•е®ҡд№үзҡ„дёҠйқўпјҢеҸҜд»Ҙз”ЁжқҘжҢҮе®ҡж–№жі•зҡ„дёҖдәӣеұһжҖ§пјҢеҰӮж–№жі•зҡ„и®ҝй—®зә§еҲ«гҖҒиҝ”еӣһеҖјзұ»еһӢгҖҒејӮеёёзұ»еһӢгҖҒжіЁйҮҠзӯүгҖӮ
                                              • еӯ—ж®өзә§еҲ«дҪңз”Ёеҹҹпјҡз”ЁдәҺжҸҸиҝ°еӯ—ж®өзҡ„жіЁи§ЈпјҢйҖҡеёёж”ҫзҪ®еңЁеӯ—ж®өе®ҡд№үзҡ„дёҠйқўпјҢеҸҜд»Ҙз”ЁжқҘжҢҮе®ҡеӯ—ж®өзҡ„дёҖдәӣеұһжҖ§пјҢеҰӮеӯ—ж®өзҡ„и®ҝй—®зә§еҲ«гҖҒй»ҳи®ӨеҖјгҖҒжіЁйҮҠзӯү

                                              дҪңз”Ё

                                              • зј–иҜ‘жЈҖжҹҘпјҡеҰӮ @Override жЈҖжҹҘж–№жі•жҳҜеҗҰжӯЈзЎ®йҮҚеҶҷ
                                              • д»Јз Ғз”ҹжҲҗпјҡеҰӮ Lombok зҡ„ @Data иҮӘеҠЁз”ҹжҲҗ getter/setter
                                              • иҝҗиЎҢж—¶еӨ„зҗҶпјҡеҰӮ Spring зҡ„ @Autowired е®һзҺ°дҫқиө–жіЁе…Ҙ

                                              еҶ…зҪ®жіЁи§Ј

                                              Java е®ҡд№үдәҶдёҖеҘ—жіЁи§ЈпјҢе…ұжңү 7 дёӘпјҢ3 дёӘеңЁ java.lang дёӯпјҢеү©дёӢ 4 дёӘеңЁ java.lang.annotation дёӯ

                                              д»Һ Java 7 ејҖе§ӢпјҢйўқеӨ–ж·»еҠ дәҶ 3 дёӘжіЁи§Ј:

                                              • @SafeVarargsпјҡJava 7 ејҖе§Ӣж”ҜжҢҒпјҢеҝҪз•Ҙд»»дҪ•дҪҝз”ЁеҸӮж•°дёәжіӣеһӢеҸҳйҮҸзҡ„ж–№жі•жҲ–жһ„йҖ еҮҪж•°и°ғз”Ёдә§з”ҹзҡ„иӯҰе‘ҠгҖӮ
                                              • @FunctionalInterfaceпјҡJava 8 ејҖе§Ӣж”ҜжҢҒпјҢж ҮиҜҶдёҖдёӘеҢҝеҗҚеҮҪж•°жҲ–еҮҪж•°ејҸжҺҘеҸЈгҖӮ
                                              • @RepeatableпјҡJava 8 ејҖе§Ӣж”ҜжҢҒпјҢж ҮиҜҶжҹҗжіЁи§ЈеҸҜд»ҘеңЁеҗҢдёҖдёӘеЈ°жҳҺдёҠдҪҝз”ЁеӨҡж¬Ў
                                              дҪңз”ЁеңЁд»Јз Ғзҡ„жіЁи§Ј
                                              • @Overrideпјҡ жЈҖжҹҘиҜҘж–№жі•жҳҜеҗҰжҳҜйҮҚеҶҷж–№жі•пјҢеҰӮжһңеҸ‘зҺ°е…¶зҲ¶зұ»пјҢжҲ–иҖ…жҳҜеј•з”Ёзҡ„жҺҘеҸЈдёӯе№¶жІЎжңүиҜҘж–№жі•ж—¶пјҢдјҡжҠҘзј–иҜ‘й”ҷиҜҜ
                                              • @Deprecatedпјҡж Үи®°иҝҮж—¶ж–№жі•гҖӮеҰӮжһңдҪҝз”ЁиҜҘж–№жі•пјҢдјҡжҠҘзј–иҜ‘иӯҰе‘ҠгҖӮ
                                              • @SuppressWarningsпјҡжҢҮзӨәзј–иҜ‘еҷЁеҺ»еҝҪз•ҘжіЁи§ЈдёӯеЈ°жҳҺзҡ„иӯҰе‘Ҡ
                                              дҪңз”ЁеңЁе…¶д»–жіЁи§Јзҡ„жіЁи§Ј(е…ғжіЁи§Ј)
                                              • @Retentionпјҡж ҮиҜҶиҝҷдёӘжіЁи§ЈжҖҺд№ҲдҝқеӯҳпјҢжҳҜеҸӘеңЁд»Јз ҒдёӯпјҢиҝҳжҳҜзј–е…Ҙclassж–Үд»¶дёӯпјҢжҲ–иҖ…жҳҜеңЁиҝҗиЎҢж—¶еҸҜд»ҘйҖҡиҝҮеҸҚе°„и®ҝй—®гҖӮ
                                              • @Documentedпјҡж Үи®°иҝҷдәӣжіЁи§ЈжҳҜеҗҰеҢ…еҗ«еңЁз”ЁжҲ·ж–ҮжЎЈдёӯгҖӮ
                                              • @Targetпјҡж Үи®°иҝҷдёӘжіЁи§Јеә”иҜҘжҳҜе“Әз§Қ Java жҲҗе‘ҳгҖӮ
                                              • @Inheritedпјҡж Үи®°иҝҷдёӘжіЁи§ЈжҳҜз»§жүҝдәҺе“ӘдёӘжіЁи§Јзұ»(й»ҳи®Ө жіЁи§Је№¶жІЎжңүз»§жүҝдәҺд»»дҪ•еӯҗзұ»)

                                              ејӮеёё

                                              Java дёӯзҡ„ејӮеёёеӨ„зҗҶжңәеҲ¶з”ЁдәҺеӨ„зҗҶзЁӢеәҸиҝҗиЎҢиҝҮзЁӢдёӯеҸҜиғҪеҸ‘з”ҹзҡ„еҗ„з§ҚејӮеёёжғ…еҶөпјҢйҖҡеёёйҖҡиҝҮ try-catch-finally иҜӯеҸҘе’Ң throw е…ій”®еӯ—жқҘе®һзҺ°

                                              Throwable жҳҜ Java иҜӯиЁҖдёӯжүҖжңүй”ҷиҜҜе’ҢејӮеёёзҡ„еҹәзұ»

                                              е®ғжңүдёӨдёӘдё»иҰҒзҡ„еӯҗзұ»пјҡError е’Ң ExceptionпјҢиҝҷдёӨдёӘзұ»еҲҶеҲ«д»ЈиЎЁдәҶ Java ејӮеёёеӨ„зҗҶдҪ“зі»дёӯзҡ„дёӨдёӘеҲҶж”Ҝ

                                              Errorзұ»

                                              Error зұ»д»ЈиЎЁйӮЈдәӣдёҘйҮҚзҡ„й”ҷиҜҜпјҢиҝҷзұ»й”ҷиҜҜйҖҡеёёжҳҜзЁӢеәҸж— жі•еӨ„зҗҶзҡ„

                                              жҜ”еҰӮпјҢOutOfMemoryError иЎЁзӨәеҶ…еӯҳдёҚи¶іпјҢStackOverflowError иЎЁзӨәж ҲжәўеҮә

                                              Exceprionзұ»

                                              Exception зұ»д»ЈиЎЁзЁӢеәҸеҸҜд»ҘеӨ„зҗҶзҡ„ејӮеёё

                                              зј–иҜ‘ж—¶ејӮеёё

                                              зј–иҜ‘ж—¶ејӮеёёпјҲChecked ExceptionпјүпјҢиҝҷзұ»ејӮеёёеңЁзј–иҜ‘ж—¶еҝ…йЎ»иў«жҳҫејҸеӨ„зҗҶпјҲжҚ•иҺ·жҲ–еЈ°жҳҺжҠӣеҮәпјү

                                              еҰӮжһңж–№жі•еҸҜиғҪжҠӣеҮәжҹҗз§Қзј–иҜ‘ж—¶ејӮеёёпјҢдҪҶжІЎжңүжҚ•иҺ·е®ғпјҲtry-catchпјүжҲ–жІЎжңүеңЁж–№жі•еЈ°жҳҺдёӯз”Ё throws еӯҗеҸҘеЈ°жҳҺе®ғпјҢйӮЈд№Ҳзј–иҜ‘е°ҶдёҚдјҡйҖҡиҝҮгҖӮдҫӢеҰӮпјҡIOExceptionгҖҒSQLException зӯү

                                              иҝҗиЎҢж—¶ејӮеёё

                                              иҝҷзұ»ејӮеёёеңЁиҝҗиЎҢж—¶жҠӣеҮәпјҢе®ғд»¬йғҪжҳҜ RuntimeException зҡ„еӯҗзұ»гҖӮеҜ№дәҺиҝҗиЎҢж—¶ејӮеёёпјҢJava зј–иҜ‘еҷЁдёҚиҰҒжұӮеҝ…йЎ»еӨ„зҗҶе®ғд»¬пјҲеҚідёҚйңҖиҰҒжҚ•иҺ·д№ҹдёҚйңҖиҰҒеЈ°жҳҺжҠӣеҮәпјү

                                              иҝҗиЎҢж—¶ејӮеёёйҖҡеёёжҳҜз”ұзЁӢеәҸйҖ»иҫ‘й”ҷиҜҜеҜјиҮҙзҡ„пјҢеҰӮ NullPointerExceptionгҖҒIndexOutOfBoundsException зӯү

                                              еҰӮжһң catch еқ—жҠӣеҮәдёҖдёӘејӮеёёпјҢиҖҢ finally еқ—дёӯд№ҹжҠӣеҮәејӮеёёпјҢйӮЈд№ҲжңҖз»ҲжҠӣеҮәзҡ„е°ҶжҳҜ finally еқ—дёӯзҡ„ејӮеёё

                                              catch еқ—дёӯзҡ„ејӮеёёдјҡиў«дёўејғпјҢиҖҢ finally еқ—дёӯзҡ„ејӮеёёдјҡиҰҶзӣ–е№¶еҗ‘дёҠдј йҖ’

                                              public class Example {
  public static void main(String[] args) {
    try {
      throw new Exception("Exception in try");
    } catch (Exception e) {
      throw new RuntimeException("Exception in catch");
    } finally {
      throw new IllegalArgumentException("Exception in finally");
    }
  }
}

                                              • дҪҶжҳҜеңЁ finally еқ—дёӯпјҢжҠӣеҮәдәҶдёҖдёӘ IllegalArgumentExceptionпјҢжңҖз»ҲзЁӢеәҸжҠӣеҮәзҡ„ејӮеёёжҳҜ finally еқ—дёӯзҡ„ IllegalArgumentException

  at com.ekko.Main.main(Main.java:12)

                                              жү§иЎҢз»“жһңпјҡ3гҖӮ

                                              try иҝ”еӣһеүҚе…Ҳжү§иЎҢ finallyпјҢз»“жһң finally йҮҢдёҚжҢүеҘ—и·ҜеҮәзүҢпјҢзӣҙжҺҘ return дәҶпјҢиҮӘз„¶д№ҹе°ұиө°дёҚеҲ° try йҮҢйқўзҡ„ return дәҶ

                                              йўҳзӣ®3

                                              return иҜӯеҸҘзҡ„еҖјдјҡдәӢе…Ҳдҝқеӯҳиө·жқҘж”ҫеҲ°иҝ”еӣһж§ҪйҮҢ

                                              йңҖиҰҒжіЁж„Ҹзҡ„жҳҜпјҢеҰӮжһңиҝ”еӣһзҡ„жҳҜеҜ№иұЎеј•з”ЁпјҢfinally дёӯдҝ®ж”№еҜ№иұЎзҡ„еұһжҖ§жҳҜдјҡз”ҹж•Ҳзҡ„

                                              public class TryDemo {
  public static void main(String[] args) {
    System.out.println(test1());
  }
  public static int test1() {
    int i = 0;
    try {
      i = 2;
      return i;
    } finally {
      i = 3;
    }
  }
}
                                              1. i = 2 - еҸҳйҮҸ i иў«иөӢеҖјдёә 2
                                              2. return i - JVM еҮҶеӨҮиҝ”еӣһпјҢжӯӨж—¶дјҡеҒҡдёӨд»¶дәӢпјҡ
                                                • и®Ўз®—иҝ”еӣһеҖјпјҡi зҡ„еҪ“еүҚеҖјжҳҜ 2
                                                • е°Ҷиҝ”еӣһеҖј 2 дҝқеӯҳеҲ°дёҖдёӘдёҙж—¶зҡ„иҝ”еӣһеҖјж§ҪдҪҚпјҲreturn slotпјүдёӯ
                                              3. жү§иЎҢ finally еқ— - i = 3пјҢеҸҳйҮҸ i иў«дҝ®ж”№дёә 3
                                              4. иҝ”еӣһд№ӢеүҚдҝқеӯҳзҡ„еҖј - иҝ”еӣһзҡ„жҳҜжӯҘйӘӨ 2 дёӯдҝқеӯҳзҡ„ 2пјҢиҖҢдёҚжҳҜеҪ“еүҚ i зҡ„еҖј 3

                                              жү§иЎҢз»“жһңпјҡ2

                                              еңЁжү§иЎҢ finally д№ӢеүҚпјҢJVM дјҡе…Ҳе°Ҷ i зҡ„з»“жһңжҡӮеӯҳиө·жқҘпјҢз„¶еҗҺ finally жү§иЎҢе®ҢжҜ•еҗҺпјҢдјҡиҝ”еӣһд№ӢеүҚжҡӮеӯҳзҡ„з»“жһңпјҢиҖҢдёҚжҳҜиҝ”еӣһ iпјҢжүҖд»ҘеҚідҪҝ i е·Із»Ҹиў«дҝ®ж”№дёә 3пјҢжңҖз»Ҳиҝ”еӣһзҡ„иҝҳжҳҜд№ӢеүҚжҡӮеӯҳиө·жқҘзҡ„з»“жһң 2
