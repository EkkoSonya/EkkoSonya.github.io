                                                                        public class Object {

  private static native void registerNatives();   //ж Үи®°дёәnativeзҡ„ж–№жі•жҳҜжң¬ең°ж–№жі•пјҢеә•еұӮжҳҜз”ұC++е®һзҺ°зҡ„
  static {
      registerNatives();   //иҝҷдёӘзұ»еңЁеҲқе§ӢеҢ–ж—¶дјҡеҜ№зұ»дёӯе…¶д»–жң¬ең°ж–№жі•иҝӣиЎҢжіЁеҶҢпјҢжң¬ең°ж–№жі•дёҚжҳҜжҲ‘д»¬SEдёӯйңҖиҰҒеӯҰд№ зҡ„еҶ…е®№пјҢжҲ‘д»¬дјҡеңЁJVMзҜҮи§Ҷйў‘ж•ҷзЁӢдёӯиҝӣиЎҢд»Ӣз»Қ
  }

  //иҺ·еҸ–еҪ“еүҚзҡ„зұ»еһӢClassеҜ№иұЎпјҢиҝҷдёӘжҲ‘д»¬дјҡеңЁжңҖеҗҺдёҖз« зҡ„еҸҚе°„дёӯиҝӣиЎҢи®Іи§ЈпјҢзӣ®еүҚжҡӮж—¶дёҚдјҡз”ЁеҲ°
  public final native Class<?> getClass();

  //иҺ·еҸ–еҜ№иұЎзҡ„е“ҲеёҢеҖјпјҢжҲ‘д»¬дјҡеңЁз¬¬дә”з« йӣҶеҗҲзұ»дёӯдҪҝз”ЁеҲ°пјҢзӣ®еүҚеҗ„дҪҚе°Ҹдјҷдјҙе°ұжҡӮж—¶зҗҶи§Јдёәдјҡиҝ”еӣһеҜ№иұЎеӯҳж”ҫзҡ„еҶ…еӯҳең°еқҖ
  public native int hashCode();

  //еҲӨж–ӯеҪ“еүҚеҜ№иұЎе’Ңз»ҷе®ҡеҜ№иұЎжҳҜеҗҰзӣёзӯүпјҢй»ҳи®Өе®һзҺ°жҳҜзӣҙжҺҘз”ЁзӯүеҸ·еҲӨж–ӯпјҢд№ҹе°ұжҳҜзӣҙжҺҘеҲӨж–ӯжҳҜеҗҰдёәеҗҢдёҖдёӘеҜ№иұЎ
  public boolean equals(Object obj) {
      return (this == obj);
  }

  //е…ӢйҡҶеҪ“еүҚеҜ№иұЎпјҢеҸҜд»Ҙе°ҶеӨҚеҲ¶дёҖдёӘе®Ңе…ЁдёҖж ·зҡ„еҜ№иұЎеҮәжқҘпјҢеҢ…жӢ¬еҜ№иұЎзҡ„еҗ„дёӘеұһжҖ§
  protected native Object clone() throws CloneNotSupportedException;

  //е°ҶеҪ“еүҚеҜ№иұЎиҪ¬жҚўдёәStringзҡ„еҪўејҸпјҢй»ҳи®Өжғ…еҶөдёӢж јејҸдёә е®Ңж•ҙзұ»еҗҚ@еҚҒе…ӯиҝӣеҲ¶е“ҲеёҢеҖј
  public String toString() {
      return getClass().getName() + "@" + Integer.toHexString(hashCode());
  }

  //е”ӨйҶ’дёҖдёӘзӯүеҫ…еҪ“еүҚеҜ№иұЎй”Ғзҡ„зәҝзЁӢпјҢжңүе…ій”Ғзҡ„еҶ…е®№пјҢжҲ‘д»¬дјҡеңЁз¬¬е…ӯз« еӨҡзәҝзЁӢйғЁеҲҶдёӯи®Іи§ЈпјҢзӣ®еүҚжҡӮж—¶дёҚдјҡз”ЁеҲ°
  public final native void notify();

  //е”ӨйҶ’жүҖжңүзӯүеҫ…еҪ“еүҚеҜ№иұЎй”Ғзҡ„зәҝзЁӢпјҢеҗҢдёҠ
  public final native void notifyAll();

  //дҪҝеҫ—жҢҒжңүеҪ“еүҚеҜ№иұЎй”Ғзҡ„зәҝзЁӢиҝӣе…Ҙзӯүеҫ…зҠ¶жҖҒпјҢеҗҢдёҠ
  public final native void wait(long timeout) throws InterruptedException;

  //еҗҢдёҠ
  public final void wait(long timeout, int nanos) throws InterruptedException {
      ...
  }

  //еҗҢдёҠ
  public final void wait() throws InterruptedException {
      ...
  }

  //еҪ“еҜ№иұЎиў«еҲӨе®ҡдёәе·Із»ҸдёҚеҶҚдҪҝз”Ёзҡ„вҖңеһғеңҫвҖқж—¶пјҢеңЁеӣһж”¶д№ӢеүҚпјҢдјҡз”ұJVMжқҘи°ғз”ЁдёҖж¬ЎжӯӨж–№жі•иҝӣиЎҢиө„жәҗйҮҠж”ҫд№Ӣзұ»зҡ„ж“ҚдҪңпјҢиҝҷеҗҢж ·дёҚжҳҜSEдёӯйңҖиҰҒеӯҰд№ зҡ„еҶ…е®№пјҢиҝҷдёӘж–№жі•жҲ‘д»¬дјҡеңЁJVMзҜҮи§Ҷйў‘ж•ҷзЁӢдёӯиҜҰз»Ҷд»Ӣз»ҚпјҢзӣ®еүҚжҡӮж—¶дёҚдјҡз”ЁеҲ°
  protected void finalize() throws Throwable { }
}

                                                                        ж–№жі•зҡ„йҮҚиҪҪжҳҜдёәжҹҗдёӘж–№жі•жҸҗдҫӣжӣҙеӨҡз§Қзұ»
                                                                        иҖҢж–№жі•зҡ„йҮҚеҶҷжҳҜиҰҶзӣ–еҺҹжңүзҡ„ж–№жі•е®һзҺ°,йҮҚеҶҷж–№жі•иҰҒжұӮдёҺзҲ¶зұ»зҡ„е®ҡд№үе®Ңе…ЁдёҖиҮҙ
                                                                        жҜ”еҰӮжҲ‘д»¬зҺ°еңЁдёҚеёҢжңӣдҪҝз”ЁObjectзұ»дёӯжҸҗдҫӣзҡ„equalsж–№жі•пјҢйӮЈд№ҲжҲ‘д»¬е°ұеҸҜд»Ҙе°Ҷе…¶йҮҚеҶҷдәҶ

                                                                        public class Person{
    ...

    @Override   //йҮҚеҶҷж–№жі•еҸҜд»Ҙж·»еҠ  @Override жіЁи§ЈпјҢжңүе…іжіЁи§ЈжҲ‘д»¬дјҡеңЁжңҖеҗҺдёҖз« иҝӣиЎҢд»Ӣз»ҚпјҢиҝҷдёӘжіЁи§Јй»ҳи®Өжғ…еҶөдёӢеҸҜд»ҘзңҒз•Ҙ
    public boolean equals(Object obj) {   //йҮҚеҶҷж–№жі•иҰҒжұӮдёҺзҲ¶зұ»зҡ„е®ҡд№үе®Ңе…ЁдёҖиҮҙ
        if(obj == null) return false;   //еҰӮжһңдј е…Ҙзҡ„еҜ№иұЎдёәnullпјҢйӮЈиӮҜе®ҡдёҚзӣёзӯү
        if(obj instanceof Person) {     //еҸӘжңүжҳҜеҪ“еүҚзұ»еһӢзҡ„еҜ№иұЎпјҢжүҚиғҪиҝӣиЎҢжҜ”иҫғпјҢиҰҒжҳҜйғҪдёҚжҳҜиҝҷдёӘзұ»еһӢиҝҳжҜ”д»Җд№Ҳ
            Person person = (Person) obj;   //е…ҲиҪ¬жҚўдёәеҪ“еүҚзұ»еһӢпјҢжҺҘзқҖжҲ‘д»¬еҜ№дёүдёӘеұһжҖ§жҢЁдёӘиҝӣиЎҢжҜ”иҫғ
            return this.name.equals(person.name) &&    //еӯ—з¬ҰдёІеҶ…е®№зҡ„жҜ”иҫғпјҢдёҚиғҪдҪҝз”Ё==пјҢеҝ…йЎ»дҪҝз”Ёequalsж–№жі•
                    this.age == person.age &&       //еҹәжң¬зұ»еһӢзҡ„жҜ”иҫғи·ҹд№ӢеүҚдёҖж ·пјҢзӣҙжҺҘ==
                    this.sex.equals(person.sex);
        }
        return false;
    }
}

                                                                        final
                                                                        еҜ№дәҺжҲҗе‘ҳеҸҳйҮҸпјҢеҲҷиЎЁзӨәеҸӘиғҪиөӢдёҖж¬ЎеҖјгҖӮеҸӘиғҪеңЁжһ„йҖ еҮҪж•°иҝӣиЎҢиөӢеҖј(еҰӮжһңжңүеҲқе§ӢеҖјпјҢжһ„йҖ еҮҪж•°д№ҹдёҚиғҪиөӢеҖј)пјҢе…¶д»–ең°ж–№дёҚиғҪдҝ®ж”№ еҜ№дәҺжҲҗе‘ҳж–№жі•пјҢдјҡйҷҗеҲ¶е…¶еӯҗзұ»дёҚе…Ғи®ёе…¶йҮҚеҶҷжүҖеҜ№еә”зҡ„жҲҗе‘ҳеҸҳйҮҸ еңЁ зұ» дёҠ еҠ  final, иЎЁзӨәиҝҷдёӘзұ»дёҚиғҪеҶҚиў«з»§жүҝдәҶ

                                                                        жҠҪиұЎзұ» abstract

                                                                        • жҠҪиұЎзұ»е…·жңү жҠҪиұЎж–№жі•пјҢжӯЈеёёе®һдҫӢеҢ–ж–№жі•жҳҜж— жі•еҲӣйҖ жҠҪиұЎзұ»зҡ„е®һдҫӢ
                                                                        • жҠҪиұЎж–№жі•жҳҜжҢҮпјҡеҸӘдҝқз•ҷж–№жі•зҡ„е®ҡд№үпјҢе№¶дёҚзј–еҶҷж–№жі•зҡ„дё»дҪ“пјҢе…·дҪ“зҡ„е®һзҺ°з”ұ еӯҗзұ» жқҘе®һзҺ°.
                                                                        • иҰҒдҪҝз”ЁжҠҪиұЎзұ»пјҢжҲ‘д»¬еҸӘиғҪеҺ»еҲӣе»әе®ғзҡ„еӯҗзұ»еҜ№иұЎгҖӮ
                                                                        • жҠҪиұЎзұ»дёҖиҲ¬еҸӘз”ЁдҪңз»§жүҝдҪҝз”ЁпјҢеҪ“з„¶пјҢжҠҪиұЎзұ»зҡ„еӯҗзұ»д№ҹеҸҜд»ҘжҳҜдёҖдёӘжҠҪиұЎзұ»
                                                                        • жҠҪиұЎж–№жі•зҡ„и®ҝй—®жқғйҷҗдёҚиғҪдёә private, еӣ дёәжҠҪиұЎж–№жі•дёҖе®ҡиҰҒз”ұеӯҗзұ»е®һзҺ°пјҢеҰӮжһңеӯҗзұ»йғҪи®ҝй—®дёҚдәҶпјҢйӮЈд№Ҳиҝҳжңүд»Җд№Ҳж„Ҹд№үе‘ўпјҹжүҖд»ҘиҜҙдёҚиғҪдёәз§ҒжңүгҖӮ
                                                                        public abstract class Person {   //йҖҡиҝҮж·»еҠ abstractе…ій”®еӯ—пјҢиЎЁзӨәиҝҷдёӘзұ»жҳҜдёҖдёӘжҠҪиұЎзұ»
    protected String name;   //еӨ§дҪ“еҶ…е®№е…¶е®һжҷ®йҖҡзұ»е·®дёҚеӨҡ
    protected int age;
    protected String sex;
    protected String profession;

    protected Person(String name, int age, String sex, String profession) {
        this.name = name;
        this.age = age;
        this.sex = sex;
        this.profession = profession;
    }

    public abstract void exam();   //жҠҪиұЎзұ»дёӯеҸҜд»Ҙе…·жңүжҠҪиұЎж–№жі•пјҢд№ҹе°ұжҳҜиҜҙиҝҷдёӘж–№жі•еҸӘжңүе®ҡд№үпјҢжІЎжңүж–№жі•дҪ“
}

                                                                        иҖҢе…·дҪ“зҡ„е®һзҺ°пјҢйңҖиҰҒз”ұеӯҗзұ»жқҘе®ҢжҲҗпјҢиҖҢдё”еҰӮжһңжҳҜеӯҗзұ»пјҢеҝ…йЎ»иҰҒе®һзҺ°жҠҪиұЎзұ»дёӯжүҖжңүжҠҪиұЎж–№жі•
                                                                        дёҚиҝҮеҰӮжһңеӯҗзұ»д№ҹжҳҜжҠҪиұЎзұ»пјҢе°ұдёҚдёҖе®ҡйңҖиҰҒе®һзҺ°гҖӮ

                                                                        public class Worker extends Person{

    public Worker(String name, int age, String sex) {
        super(name, age, sex, "е·Ҙдәә");
    }

    @Override
    public void exam() {   //еӯҗзұ»еҝ…йЎ»иҰҒе®һзҺ°жҠҪиұЎзұ»жүҖжңүзҡ„жҠҪиұЎж–№жі•пјҢиҝҷжҳҜејәеҲ¶иҰҒжұӮзҡ„пјҢеҗҰеҲҷдјҡж— жі•йҖҡиҝҮзј–иҜ‘
        System.out.println("жҲ‘жҳҜе·ҘдәәпјҢеҒҡйўҳжҲ‘е№¶дёҚж“…й•ҝпјҢеҸӘиғҪеҫ—еҲ° D");
    }
}

                                                                        еҸ‘зҺ°еҜ№дәҺ жҠҪиұЎзұ» дёӯе®ҡд№үзҡ„ жҠҪиұЎж–№жі•пјҢе…¶еӯҗзұ»зҡ„еҜ№еә”зҡ„ж–№жі•зҡ„и®ҝй—®жқғйҷҗйңҖиҰҒй«ҳдәҺжҠҪиұЎзұ»дёӯзҡ„ж–№жі•пјҢдё”еҗҢж ·дёҚиғҪдҪҝз”Ё privateгҖӮ
                                                                        еҚі еҰӮжһңжҠҪиұЎж–№жі•еңЁжҠҪиұЎзұ»е®ҡд№үзҡ„жҳҜ public, еӯҗзұ»еҜ№еә”еҝ…йЎ»жҳҜ public дёҚиғҪжҳҜ protected; иҖҢеҰӮжһңжҠҪиұЎзұ»е®ҡд№үзҡ„жҳҜ protectedпјҢеӯҗзұ»д№ҹеҸҜд»Ҙе®ҡд№ү public

                                                                        жҺҘеҸЈ interface

                                                                        • жҺҘеҸЈз”ҡиҮіжҜ”жҠҪиұЎзұ»иҝҳжҠҪиұЎпјҢд»–еҸӘд»ЈиЎЁжҹҗдёӘзЎ®еҲҮзҡ„еҠҹиғҪпјҒд№ҹе°ұжҳҜеҸӘеҢ…еҗ«ж–№жі•зҡ„е®ҡд№үпјҢз”ҡиҮійғҪдёҚжҳҜдёҖдёӘзұ»пјҒ
                                                                        • жҺҘеҸЈдёҖиҲ¬еҸӘд»ЈиЎЁжҹҗдәӣеҠҹиғҪзҡ„жҠҪиұЎпјҢжҺҘеҸЈеҢ…еҗ«дәҶдёҖдәӣеҲ—ж–№жі•зҡ„е®ҡд№үпјҢзұ»еҸҜд»Ҙе®һзҺ°иҝҷдёӘжҺҘеҸЈпјҢиЎЁзӨәзұ»ж”ҜжҢҒжҺҘеҸЈд»ЈиЎЁзҡ„еҠҹиғҪпјҲзұ»дјјдәҺдёҖдёӘжҸ’д»¶пјҢеҸӘиғҪдҪңдёәдёҖдёӘйҷ„еұһеҠҹиғҪеҠ еңЁдё»дҪ“дёҠпјҢеҗҢж—¶е…·дҪ“е®һзҺ°иҝҳйңҖиҰҒз”ұдё»дҪ“жқҘе®һзҺ°пјү
                                                                        • е®һйҷ…дёҠжҺҘеҸЈзҡ„зӣ®ж Үе°ұжҳҜе°Ҷзұ»жүҖе…·жңүжҹҗдәӣзҡ„иЎҢдёәжҠҪиұЎеҮәжқҘгҖӮ
                                                                        • еҸҜд»ҘзҗҶи§Јдёә жҺҘеҸЈ зӣёеҪ“дәҺ еҸӘжңүжҠҪиұЎзұ»дёӯзҡ„жҠҪиұЎж–№жі•пјҢз”ҡиҮійғҪдёҚжҳҜдёҖдёӘзұ»дәҶгҖӮ
                                                                        • жҺҘеҸЈйҮҢеҸӘиғҪе®ҡд№үеҜ№еә”зҡ„жҠҪиұЎж–№жі•пјҢдёҚиҝҮеҸҜд»ҘзңҒз•Ҙ abstract е®ҡд№ү е№¶дё”й»ҳи®ӨеңЁзұ»дёӯе®һзҺ°зҡ„жқғйҷҗжҳҜ public
                                                                        • е®ҡд№үжҺҘеҸЈ interface
                                                                        • е®һзҺ°жҺҘеҸЈ implements
                                                                        • жҺҘеҸЈеҸҜд»Ҙе®һзҺ°еҫҲеӨҡдёӘпјҢеҸӘйңҖиҰҒз”Ё йҖ—еҸ· йҡ”ејҖеҚіеҸҜпјҢзұ»еҸӘиғҪз»§жүҝдёҖдёӘгҖҒ
                                                                        • жүҖд»ҘиҜҙжңүдәӣдәәиҜҙжҺҘеҸЈе…¶е®һе°ұжҳҜJavaдёӯзҡ„еӨҡз»§жүҝпјҢдҪҶжҳҜжҲ‘дёӘдәәи®Өдёәиҝҷз§ҚиҜҙжі•жҳҜй”ҷзҡ„пјҢе®һйҷ…дёҠе®һзҺ°жҺҘеҸЈжӣҙеғҸжҳҜдёҖдёӘзұ»зҡ„еҠҹиғҪеҲ—иЎЁпјҢдҪңдёәйҷ„еҠ еҠҹиғҪеӯҳеңЁпјҢдёҖдёӘзұ»еҸҜд»Ҙйҷ„еҠ еҫҲеӨҡдёӘеҠҹиғҪпјҢжҺҘеҸЈзҡ„дҪҝз”Ёе’Ңз»§жүҝзҡ„жҰӮеҝөжңүдёҖе®ҡзҡ„еҮәе…ҘпјҢйЎ¶еӨҡиҜҙжҳҜеӨҡз»§жүҝзҡ„дёҖз§Қжӣҝд»Јж–№жЎҲгҖӮ
                                                                        • java8ејҖе§ӢпјҢжҺҘеҸЈдёӯзҡ„ж–№жі•еҸҜд»ҘеӯҳеңЁй»ҳи®Өе®һзҺ°пјҢdefault еҰӮжһңж–№жі•еңЁжҺҘеҸЈдёӯеӯҳеңЁй»ҳи®Өе®һзҺ°пјҢйӮЈд№Ҳе®һзҺ°зұ»дёӯдёҚејәеҲ¶иҰҒжұӮиҝӣиЎҢе®һзҺ°гҖӮ
                                                                        • жҺҘеҸЈдёҚеҗҢдәҺзұ»пјҢжҺҘеҸЈдёӯдёҚе…Ғи®ёеӯҳеңЁжҲҗе‘ҳеҸҳйҮҸе’ҢжҲҗе‘ҳж–№жі•пјҢдҪҶжҳҜеҸҜд»ҘеӯҳеңЁйқҷжҖҒеҸҳйҮҸе’ҢйқҷжҖҒж–№жі• жҺҘеҸЈдёӯе®ҡд№үзҡ„йқҷжҖҒеҸҳйҮҸеҸӘиғҪжҳҜpublic static finalзҡ„
                                                                          жҺҘеҸЈдёӯе®ҡд№үзҡ„йқҷжҖҒж–№жі•д№ҹеҸӘиғҪжҳҜpublicзҡ„ иҝҷдәӣеҸҜд»ҘзңҒ зӣҙжҺҘint a = 1 static void test()иҝҷз§ҚеҚіеҸҜ и·ҹжҷ®йҖҡзҡ„зұ»дёҖж ·пјҢжҲ‘д»¬еҸҜд»ҘзӣҙжҺҘйҖҡиҝҮжҺҘеҸЈеҗҚ.зҡ„ж–№ејҸдҪҝз”ЁйқҷжҖҒеҶ…е®№
                                                                        • жҺҘеҸЈжҳҜеҸҜд»Ҙз»§жүҝ (extends) иҮӘе…¶д»–жҺҘеҸЈзҡ„, е№¶дё”жҺҘеҸЈжІЎжңүз»§жүҝж•°йҮҸйҷҗеҲ¶пјҢжҺҘеҸЈж”ҜжҢҒеӨҡз»§жүҝ жҺҘеҸЈзҡ„з»§жүҝзӣёеҪ“дәҺжҳҜеҜ№жҺҘеҸЈеҠҹиғҪзҡ„иһҚеҗҲзҪўдәҶ
                                                                        • жҺҘеҸЈзҡ„й»ҳи®Өж–№жі•жҳҜдҝқеә•зҡ„пјҢеҸӘиҰҒдёҖдёӘзұ»зҡ„зҲ¶зұ»жҲ–иҖ…иҮӘиә«жңүеҜ№еә”ж–№жі•пјҢе°ұдёҚдјҡжү§иЎҢжҺҘеҸЈзҡ„й»ҳи®Өж–№жі•
                                                                        • жҺҘеҸЈдёӯеҰӮжһңе®ҡд№үдәҶдёҺ Object еҗҢеҗҚзҡ„ж–№жі•пјҢдёҚиғҪдҪҝз”Ёй»ҳи®ӨпјҢеӣ дёәе…¶д»–зұ»е°ұз®—з»§жүҝиҝҷдёӘжҺҘеҸЈпјҢз”ұдәҺзұ»жң¬иә«йғҪжҳҜз»§жүҝ Object зҡ„пјҢиҝҷдёӘй»ҳи®Өж–№жі•жІЎжңүд»»дҪ•дҪңз”Ё

                                                                        жҜ”еҰӮиҜҙпјҢеҜ№дәҺдәәзұ»зҡ„дёҚеҗҢеӯҗзұ»пјҢеӯҰз”ҹе’ҢиҖҒеёҲжқҘиҜҙпјҢд»–д»¬йғҪе…·жңүеӯҰд№ иҝҷдёӘиғҪеҠӣпјҢж—ўз„¶йғҪжңүпјҢйӮЈд№ҲжҲ‘д»¬е°ұеҸҜд»Ҙе°ҶеӯҰд№ иҝҷдёӘиғҪеҠӣпјҢжҠҪиұЎжҲҗжҺҘеҸЈжқҘиҝӣиЎҢдҪҝз”ЁпјҢеҸӘиҰҒжҳҜе®һзҺ°иҝҷдёӘжҺҘеҸЈзҡ„зұ»пјҢйғҪжңүеӯҰд№ зҡ„иғҪеҠӣпјҡ

                                                                        package java.lang;

public interface Cloneable {    //иҝҷдёӘжҺҘеҸЈдёӯд»Җд№ҲйғҪжІЎе®ҡд№ү
}

                                                                        е…·дҪ“е®һзҺ°е…ӢйҡҶ:

                                                                        public class Student extends Person implements Study, Cloneable {   //йҰ–е…Ҳе®һзҺ°CloneableжҺҘеҸЈпјҢиЎЁзӨәиҝҷдёӘзұ»е…·жңүе…ӢйҡҶзҡ„еҠҹиғҪ
    public Student(String name, int age, String sex) {
        super(name, age, sex, "еӯҰз”ҹ");
    }

    @Override
    public Object clone() throws CloneNotSupportedException {   //жҸҗеҚҮcloneж–№жі•зҡ„и®ҝй—®жқғйҷҗ
        return super.clone();   //еӣ дёәеә•еұӮжҳҜC++е®һзҺ°пјҢжҲ‘д»¬зӣҙжҺҘи°ғз”ЁзҲ¶зұ»зҡ„е®һзҺ°е°ұеҸҜд»ҘдәҶ
    }

    @Override
    public void study() {
        System.out.println("жҲ‘дјҡеӯҰд№ пјҒ");
    }
}

                                                                        е…ӢйҡҶе®һзҺ°пјҡ

                                                                        public static void main(String[] args) throws CloneNotSupportedException {  //иҝҷйҮҢеҗ‘дёҠжҠӣеҮәдёҖдёӢејӮеёёпјҢиҝҳжІЎеӯҰејӮеёёпјҢжүҖд»ҘиҜҙз…§зқҖеҶҷе°ұиЎҢдәҶ
    Student student = new Student("е°ҸжҳҺ", 18, "з”·");
    Student clone = (Student) student.clone();   //и°ғз”Ёcloneж–№жі•пјҢеҫ—еҲ°дёҖдёӘе…ӢйҡҶзҡ„еҜ№иұЎ
    System.out.println(student);
    System.out.println(clone);
    System.out.println(student == clone);
}

    RUNNING, STUDY, SLEEP;    
    //зӣҙжҺҘеҶҷжҜҸдёӘзҠ¶жҖҒзҡ„еҗҚеӯ—еҚіеҸҜпјҢжңҖеҗҺйқўеҲҶеҸ·еҸҜд»ҘдёҚжү“пјҢдҪҶжҳҜжҺЁиҚҗжү“дёҠ
}

                                                                        дҪҝз”Ёжһҡдёҫзұ»д№ҹйқһеёёж–№дҫҝпјҢе°ұеғҸдҪҝз”Ёжҷ®йҖҡзұ»еһӢйӮЈж ·пјҡ

                                                                        public class Student {

    private Status status;   //зҠ¶жҖҒпјҢеҸҜд»ҘжҳҜи·‘жӯҘгҖҒеӯҰд№ гҖҒзқЎи§үиҝҷдёүдёӘд№Ӣдёӯзҡ„е…¶дёӯдёҖз§Қ

    public Status getStatus() {
        return status;
    }

    public void setStatus(Status status) {
        this.status = status;
    }
}

                                                                        жһҡдёҫзұ»еһӢдҪҝз”Ёиө·жқҘе°ұйқһеёёж–№дҫҝдәҶпјҢе…¶е®һжһҡдёҫзұ»еһӢзҡ„жң¬иҙЁе°ұжҳҜдёҖдёӘжҷ®йҖҡзҡ„зұ»пјҢдҪҶжҳҜе®ғз»§жүҝиҮӘEnumзұ»пјҢжҲ‘д»¬е®ҡд№үзҡ„жҜҸдёҖдёӘзҠ¶жҖҒе…¶е®һе°ұжҳҜдёҖдёӘpublic static finalзҡ„Statusзұ»еһӢжҲҗе‘ҳеҸҳйҮҸпјҡ

                                                                        //иҝҷйҮҢдҪҝз”Ёjavapе‘Ҫд»ӨеҜ№classж–Үд»¶иҝӣиЎҢеҸҚзј–иҜ‘еҫ—еҲ° Compiled from "Status.java"
public final class com.test.Status extends java.lang.Enum<com.test.Status> {
  public static final com.test.Status RUNNING;
  public static final com.test.Status STUDY;

  public static com.test.Status valueOf(java.lang.String);
  static {};
}

                                                                        жһҡдёҫзұ»еһӢжҳҜжҷ®йҖҡзҡ„зұ»пјҢйӮЈд№ҲжҲ‘д»¬д№ҹеҸҜд»Ҙз»ҷжһҡдёҫзұ»еһӢж·»еҠ зӢ¬жңүзҡ„жҲҗе‘ҳж–№жі•пјҡ

                                                                        public enum Status {
    RUNNING("зқЎи§ү"), STUDY("еӯҰд№ "), SLEEP("зқЎи§ү");   //ж— еҸӮжһ„йҖ ж–№жі•иў«иҰҶзӣ–пјҢеҲӣе»әжһҡдёҫйңҖиҰҒж·»еҠ еҸӮж•°пјҲжң¬иҙЁе°ұжҳҜи°ғз”Ёзҡ„жһ„йҖ ж–№жі•пјү

    private final String name;    //жһҡдёҫзҡ„жҲҗе‘ҳеҸҳйҮҸ
    Status(String name){    //иҰҶзӣ–еҺҹжңүжһ„йҖ ж–№жі•пјҲй»ҳи®ӨprivateпјҢеҸӘиғҪеҶ…йғЁдҪҝз”ЁпјҒпјү
        this.name = name;
    }

    public String getName() {   //иҺ·еҸ–е°ҒиЈ…зҡ„жҲҗе‘ҳеҸҳйҮҸ
        return name;
    }
}
